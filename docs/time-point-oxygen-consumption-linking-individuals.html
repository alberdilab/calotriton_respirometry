  filter(time_point!="0") %>%
  ggplot(aes(y = o2_normalized, x = interaction(time_point, individual), color = time_point, fill = time_point)) +
  geom_point(position = position_jitterdodge(jitter.width = 0.2, dodge.width = 0.8), alpha = 0.5) +
  facet_grid(temperature~factor(elevation)+population, scale="free") +
  geom_line(aes(group=individual)) +
  theme_minimal()+
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1, size=0))+
  labs(x = "Time-point", y = "Normalized o2 consumption (ml/s/g)", color = "Time-point", fill = "Time-point")
